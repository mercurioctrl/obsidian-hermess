## Descripción

Se debe generar un reporte en xmlsx idéntico al archivo adjunto en esta historia

```
GET {{API_URL}}/v1/reports/billingreport?clientId={}&categoryId={}&branch={}&date={}&search={}&brandId={}
```



Filtros

- Cliente (Nombre o id)


- Familia (Id)


- Marca (brandId)


- Filtro (Descripcion parcial del nombre de producto)


- Sucursal (10,02)


- Fecha



```sql
SELECT a.ID_PRODUCTO AS SKU
    , f.dfecfac AS FECHA_FACTURA
    , r.FECHA_LIQUIDACION AS FECHA_LIQUIDACION
    , r.FECHA_LIQUIDACION
    , a.cdetalle AS TITULO
    , CAST(l.ncanent AS INT) AS CANTIDAD
    , REPLACE(CAST(l.npreunit - l.npreunit * l.ndto / 100 AS MONEY), '.', ',') AS PRECIO
    , f.cfactura AS FACTURA
    , t.cnumalb AS REMITO_F
    , p.cnompobl AS CIUDAD
    , pr.Descripcion AS PROVINCIA
    , t.ccodage AS AGENTE
    , c.cnomcli AS CLIENTE
    , c.cdnicif AS CUIT
    , ag.capeage AS VENDEDOR_NOMBRE
    , ag.cnbrage AS VENDEDOR_APELLIDO
    , c.ccodcli AS CLIENTE_ID
    , fam.cnomfam AS CATEGORIA
    , m.referencia AS MARCA
    , c.cdircli AS DIRECCION
FROM NewBytes_DBF.dbo.albclil l
LEFT JOIN NewBytes_DBF.dbo.articulo a
    ON l.ID_Articulo = a.ID_Articulo
LEFT JOIN NewBytes_DBF.dbo.familias fam
    ON fam.ccodfam = a.ccodfam
LEFT JOIN NB_WEB.dbo.marcas m
    ON m.id = a.Id_Marca
LEFT JOIN NewBytes_DBF.dbo.albclit t
    ON t.ID_NROREMCLI_ENC = l.ID_NROREMCLI_ENC
LEFT JOIN NewBytes_DBF.dbo.clientes c
    ON t.ID_CLIENTE = c.ID_CLIENTE
LEFT JOIN NewBytes_DBF.dbo.agentes ag
    ON t.ccodage = ag.ccodage
LEFT JOIN NewBytes_DBF.DBO.poblac p
    ON c.ccodpobl = p.ccodpobl
        AND p.sitio = 1
LEFT JOIN NewBytes_DBF.dbo.FP_Provincias pr
    ON pr.Id_Provincia = p.ncodprov
LEFT JOIN [NewBytes_DBF].[dbo].[FP_FactWebCliEncabezado] f
    ON t.ID_NROREMCLI_ENC = f.ID_NROREMCLI_ENC
LEFT JOIN NEW_BYTES.dbo.MS_REMITO_CABECERA r
    ON t.cnumalb = r.REMITO_FP
        AND t.cnumsuc = r.SUCURSAL_REMITO
WHERE t.ntipoalb > 1
    AND fam.ID_FAMILIA NOT IN (65)
    AND r.FECHA_LIQUIDACION BETWEEN '01-03-2024' AND '13-03-2024 23:59'
    AND F.dfecfac IS NULL
    AND (
        f.lanulada <> 1
        OR f.lanulada IS NULL
        )
    AND (
        f.NTIPODOCU <> 2
        OR f.NTIPODOCU IS NULL
        )
    AND (
        t.ccodcli <> 6
        AND t.ccodcli <> 29719
        AND t.ccodcli <> 33181
        )
GROUP BY a.ID_PRODUCTO
    , f.dfecfac
    , r.FECHA_LIQUIDACION
    , a.cdetalle
    , l.ncanent
    , l.npreunit
    , l.ndto
    , f.cfactura
    , t.cnumalb
    , p.cnompobl
    , pr.Descripcion
    , t.ccodage
    , c.cnomcli
    , c.cdnicif
    , ag.capeage
    , ag.cnbrage
    , c.ccodcli
    , fam.cnomfam
    , m.referencia
    , c.cdircli
    , r.FECHA_LIQUIDACION

UNION

SELECT a.ID_PRODUCTO AS SKU
    , f.dfecfac AS FECHA_FACTURA
    , r.FECHA_LIQUIDACION AS FECHA_LIQUIDACION
    , r.FECHA_LIQUIDACION
    , a.cdetalle AS TITULO
    , CAST(l.ncanent AS INT) AS CANTIDAD
    , REPLACE(CAST(l.npreunit - l.npreunit * l.ndto / 100 AS MONEY), '.', ',') AS PRECIO
    , f.cfactura AS FACTURA
    , t.cnumalb AS REMITO_F
    , p.cnompobl AS CIUDAD
    , pr.Descripcion AS PROVINCIA
    , t.ccodage AS AGENTE
    , c.cnomcli AS CLIENTE
    , c.cdnicif AS CUIT
    , ag.capeage AS VENDEDOR_NOMBRE
    , ag.cnbrage AS VENDEDOR_APELLIDO
    , c.ccodcli AS CLIENTE_ID
    , fam.cnomfam AS CATEGORIA
    , m.referencia AS MARCA
    , c.cdircli AS DIRECCION
FROM NewBytes_DBF.dbo.albclil l
LEFT JOIN NewBytes_DBF.dbo.articulo a
    ON l.ID_Articulo = a.ID_Articulo
LEFT JOIN NewBytes_DBF.dbo.familias fam
    ON fam.ccodfam = a.ccodfam
LEFT JOIN NB_WEB.dbo.marcas m
    ON m.id = a.Id_Marca
LEFT JOIN NewBytes_DBF.dbo.albclit t
    ON t.ID_NROREMCLI_ENC = l.ID_NROREMCLI_ENC
LEFT JOIN NewBytes_DBF.dbo.clientes c
    ON t.ID_CLIENTE = c.ID_CLIENTE
LEFT JOIN NewBytes_DBF.dbo.agentes ag
    ON t.ccodage = ag.ccodage
LEFT JOIN NewBytes_DBF.DBO.poblac p
    ON c.ccodpobl = p.ccodpobl
        AND p.sitio = 1
LEFT JOIN NewBytes_DBF.dbo.FP_Provincias pr
    ON pr.Id_Provincia = p.ncodprov
LEFT JOIN [NewBytes_DBF].[dbo].[FP_FactWebCliEncabezado] f
    ON t.ID_NROREMCLI_ENC = f.ID_NROREMCLI_ENC
LEFT JOIN NEW_BYTES.dbo.MS_REMITO_CABECERA r
    ON t.cnumalb = r.REMITO_FP
        AND t.cnumsuc = r.SUCURSAL_REMITO
WHERE t.ntipoalb > 1
    AND fam.ID_FAMILIA NOT IN (65)
    AND F.dfecfac BETWEEN '01-03-2022' AND '13-03-2024 23:59'
    AND F.dfecfac IS NOT NULL
    AND (
        f.lanulada <> 1
        OR f.lanulada IS NULL
        )
    AND (
        f.NTIPODOCU <> 2
        OR f.NTIPODOCU IS NULL
        )
    AND (
        t.ccodcli <> 6
        AND t.ccodcli <> 29719
        AND t.ccodcli <> 33181
        )
GROUP BY a.ID_PRODUCTO
    , f.dfecfac
    , r.FECHA_LIQUIDACION
    , a.cdetalle
    , l.ncanent
    , l.npreunit
    , l.ndto
    , f.cfactura
    , t.cnumalb
    , p.cnompobl
    , pr.Descripcion
    , t.ccodage
    , c.cnomcli
    , c.cdnicif
    , ag.capeage
    , ag.cnbrage
    , c.ccodcli
    , fam.cnomfam
    , m.referencia
    , c.cdircli
    , r.FECHA_LIQUIDACION
ORDER BY r.FECHA_LIQUIDACION ASC
```
